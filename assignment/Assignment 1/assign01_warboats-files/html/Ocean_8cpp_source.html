179  /*!*************************************************************************
180  \fn ShotResult TakeShot(Ocean &ocean, const Point &coordinate)
181 
182  \brief
183  Client calls this in an attempt to hit one of the boats.
184 
185  The coordinate parameter indicates where the client is attempting to
186  strike. There are several possible results: Hit, Miss, Sunk, Duplicate,
187  or Illegal. Hit, Miss, and Duplicate are obvious. Sunk is returned when
188  a shot hits the last undamaged part of a boat. Sunk implies Hit.
189  Illegal is any coordinate that is outside of the ocean (e.g. x or y '=
190  less than 0 or outside the range).
191 
192  \param ocean
193  The ocean to fire at.
194 
195  \param coordinate
196  The coordinate to the user is trying to attack in the given ocean.
197 
198  \return
199  The result of the shot. Can be any of the following:
200  srHIT, srMISS, srDUPLICATE, srSUNK, srILLEGAL
201  ***************************************************************************/
202  ShotResult TakeShot(Ocean &ocean, const Point &coordinate)
203  {
204  // If point is not in ocean.
205  if(!IsInOcean(coordinate, &ocean))
206  {
207  // return illegal.
208  return srILLEGAL;
209  }
210  // Get the point in ocean that was fired upon.
211  int& value = PointInOcean(ocean, coordinate.x, coordinate.y);
212 
213  // If the shot is a miss.
214  if(value == 0)
215  {
216  // Increment Misses.
217  ocean.stats.misses++;
218 
219  // Set position to blown up.
220  value = -1;
221 
222  //return miss shot result.
223  return srMISS;
224  }
225 
226  // A shot is a duplicate if its a hit, or a -1. A hit is above 100.
227  if(value == -1 || value >= 100)
228  {
229  // Increment duplicates
230  ocean.stats.duplicates++;
231 
232  // Return Duplicate shot result
233  return srDUPLICATE;
234  }
235 
236  // If we are here, we must have hit a boat. Increment hit.
237  ocean.stats.hits++;
238 
239  // Bool to record of the boat has sunk.
240  bool hasSunk = false;
241 
242  // Find the boat that was hit and increment it too. Value is ID.
243  for(int i = 0; i < ocean.num_boats; i++)
244  {
245  // Find the boat with the corresponding ID.
246  if(ocean.boats[i].ID == value)
247  {
248  // Increment the hits the boat has taken.
249  ocean.boats[i].hits++;
250 
251  // If the boat is sinking, we record it and deal with it later.
252  if(ocean.boats[i].hits >= BOAT_LENGTH)
253  {
254  hasSunk = true;
255  }
256  break;
257  }
258  }
259 
260  //Update position on ocean to boat hit
261  value += 100;
262 
263  // Now we deal if the boat has sunk.
264  if(hasSunk)
265  {
266  // Increment sunk
267  ocean.stats.sunk++;
268 
269  // return Sunk shot result
270  return srSUNK;
271  }
272 
273  // Return Hit shot result
274  return srHIT;
275  }

277  /*!*************************************************************************
278  \fn BoatPlacement PlaceBoat(Ocean &ocean, const Boat& boat)
279 
280  \brief
281  Given an ocean and a boat, places the boat in the ocean.
282 
283  The client calls this to place boats in the ocean. The client will
284  create a boat and pass it to the function. The function must ensure
285  that the boat will "fit" in the ocean. Do this by checking the location
286  where it is to be placed, the orientation (horizontal/vertical), and
287  whether or not it will overlap with another boat or stick outside of
288  the ocean. The return value indicates whether or not the boat could be
289  placed in the ocean. See WarBoats.h for valid values of BoatPlacement.
290 
291  \param ocean
292  The ocean to place the boat at.
293 
294  \param boat
295  The boat which to place in the ocean.
296 
297  \return
298  Whether the boat was successfully placed in the ocean. Can either be
299  bpACCEPTED, or bpREJECTED.
300  ***************************************************************************/
301  BoatPlacement PlaceBoat(Ocean &ocean, const Boat& boat)
302  {
303  // Check if the left/top of the boat is not in the ocean.
304  if(!IsInOcean(boat.position, &ocean))
305  {
306  // Reject the placement.
307  return bpREJECTED;
308  }
309 
310  // If boat is horizontal
311  if(boat.orientation == oHORIZONTAL)
312  {
313  // Check if right most position is not in map.
314  if(!IsInOcean(boat.position.x + BOAT_LENGTH - 1, boat.position.y,
315  &ocean))
316  {
317  // Reject
318  return bpREJECTED;
319  }
320  }
321  // Boat is vertical.
322  else
323  {
324  // Check the if bottom point is not in map.
325  if(!IsInOcean(boat.position.x, boat.position.y + BOAT_LENGTH - 1,
326  &ocean))
327  {
328  // Reject
329  return bpREJECTED;
330  }
331  }
332 
333  // Now check if the boats intersect with other boats.
334  for (int i = 0; i < BOAT_LENGTH; i++)
335  {
336  int x = boat.position.x; /* Easy way to get/set position x.*/
337  int y = boat.position.y; /* Easy way to get/set position y.*/
338 
339  // If orientation is horizontal, we need to i to go towards the right.
340  if (boat.orientation == oHORIZONTAL)
341  x += i;
342  else // else if vertical, we need to i to go towards the bottom.
343  y += i;
344 
345  // If the point on the ocean is not empty.
346  if (PointInOcean(ocean, x, y) != 0)
347  {
348  // Return reject.
349  return bpREJECTED;
350  }
351  }
352 
353  // Now the place is clear to set the boat.
354  for(int i = 0; i < BOAT_LENGTH; i++)
355  {
356  int x = boat.position.x; /* Easy way to get/set position x.*/
357  int y = boat.position.y; /* Easy way to get/set position y.*/
358 
359  // If orientation is horizontal, we need to i to go towards the right.
360  if(boat.orientation == oHORIZONTAL)
361  x += i;
362  else // else if vertical, we need to i to go towards the bottom.
363  y += i;
364 
365  // If the point on the ocean is empty.
366  if(PointInOcean(ocean, x, y) == 0)
367  {
368  // Set the point on the map to the ID.
369  PointInOcean(ocean, x, y) = boat.ID;
370  }
371  }
372  static int index = 0; // Counter for boats array.
373 
374  // All went well, boat has been set. Add to array and retun accepted.
375  ocean.boats[index] = boat;
376  ocean.boats[index++].hits = 0; // Initialize the hit value if it isn't.
377  return bpACCEPTED;
378  }
